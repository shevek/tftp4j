005package org.anarres.tftp.protocol.engine;
006
007import java.io.IOException;
008import javax.annotation.Nonnegative;
009import javax.annotation.Nonnull;
010import javax.annotation.PostConstruct;
011import javax.annotation.PreDestroy;
012import org.anarres.tftp.protocol.resource.TftpDataProvider;
013
014/**
015 *
016 * @author shevek
017 */
018public abstract class AbstractTftpServer {
019
020    public static final int DEFAULT_SERVER_PORT = 69;
021
022    private final TftpDataProvider dataProvider;
023    private final int port;
024
025    public AbstractTftpServer(@Nonnull TftpDataProvider dataProvider, @Nonnegative int port) {
026        this.dataProvider = dataProvider;
027        this.port = port;
028    }
029
030    @Nonnull
031    public TftpDataProvider getDataProvider() {
032        return dataProvider;
033    }
034
035    @Nonnegative
036    public int getPort() {
037        return port;
038    }
039
040    @PostConstruct
041    public abstract void start() throws IOException, InterruptedException;
042
043    @PreDestroy
044    public abstract void stop() throws IOException, InterruptedException;
045}
