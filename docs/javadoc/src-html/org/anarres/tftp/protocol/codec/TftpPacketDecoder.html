005package org.anarres.tftp.protocol.codec;
006
007import java.net.SocketAddress;
008import java.nio.ByteBuffer;
009import javax.annotation.Nonnull;
010import org.anarres.tftp.protocol.packet.TftpAckPacket;
011import org.anarres.tftp.protocol.packet.TftpDataPacket;
012import org.anarres.tftp.protocol.packet.TftpErrorPacket;
013import org.anarres.tftp.protocol.packet.TftpOpcode;
014import org.anarres.tftp.protocol.packet.TftpPacket;
015import org.anarres.tftp.protocol.packet.TftpReadRequestPacket;
016import org.anarres.tftp.protocol.packet.TftpWriteRequestPacket;
017import org.slf4j.Logger;
018import org.slf4j.LoggerFactory;
019
020/**
021 *
022 * @author shevek
023 */
024public class TftpPacketDecoder {
025
026    private static final Logger LOG = LoggerFactory.getLogger(TftpPacketDecoder.class);
027
028    @Nonnull
029    public TftpPacket decode(@Nonnull SocketAddress remoteAddress, @Nonnull ByteBuffer buf) {
030        TftpOpcode opcode = TftpOpcode.forCode(buf.getShort());
031        TftpPacket packet;
032        switch (opcode) {
033            case RRQ:
034                packet = new TftpReadRequestPacket();
035                break;
036            case WRQ:
037                packet = new TftpWriteRequestPacket();
038                break;
039            case DATA:
040                packet = new TftpDataPacket();
041                break;
042            case ACK:
043                packet = new TftpAckPacket();
044                break;
045            case ERROR:
046                packet = new TftpErrorPacket();
047                break;
048            case ACK_WITH_OPTIONS:
049            default:
050                throw new IllegalStateException("Unknown TftpOpcode in decoder: " + opcode);
051        }
052        packet.setRemoteAddress(remoteAddress);
053        packet.fromWire(buf);
054        if (buf.position() < buf.limit()) {
055            LOG.warn("Discarded " + (buf.limit() - buf.position()) + " trailing bytes in TFTP packet: " + buf);
056            buf.position(buf.limit());
057        }
058        return packet;
059    }
060}
